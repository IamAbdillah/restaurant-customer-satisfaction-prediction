import os
import json
import warnings
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

from sklearn.model_selection import train_test_split, StratifiedKFold, RandomizedSearchCV
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import OneHotEncoder, StandardScaler
from sklearn.impute import SimpleImputer
from sklearn.metrics import (
    accuracy_score, precision_score, recall_score, f1_score,
    roc_auc_score, average_precision_score, confusion_matrix,
    classification_report, precision_recall_curve
)
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.inspection import permutation_importance
from xgboost import XGBClassifier
from catboost import CatBoostClassifier
import joblib

warnings.filterwarnings("ignore")
sns.set(style="whitegrid")
print("Versions -> numpy:", np.__version__, "| pandas:", pd.__version__)

# ---------- 2) LOAD DATA ----------
# Upload your CSV in Colab, then set this filename
DATA_PATH = "/content/restaurant_customer_satisfaction.csv"
df = pd.read_csv(DATA_PATH)

print("Shape:", df.shape)
print("\nColumns:", list(df.columns))
print("\nMissing values:\n", df.isna().sum())
print("\nTarget distribution:\n", df["HighSatisfaction"].value_counts())
print("\nTarget rate:", round(df["HighSatisfaction"].mean(), 4))

# ---------- 3) QUICK EDA ----------
# 3.1 Numeric correlation heatmap (global data understanding)
plt.figure(figsize=(12, 8))
num_df = df.select_dtypes(include=np.number)
sns.heatmap(
    num_df.corr(),
    cmap="coolwarm",
    center=0,
    annot=True,
    fmt=".2f",
    linewidths=0.5
)
plt.title("Correlation Heatmap (Numerical Features)")
plt.tight_layout()
plt.show()

# 3.2 Categorical satisfaction-rate check (optional quick view)
cat_cols = [c for c in df.columns if df[c].dtype == "object"]
for c in cat_cols:
    temp = df.groupby(c)["HighSatisfaction"].mean().sort_values(ascending=False)
    print(f"\nSatisfaction rate by {c}:")
    print((temp * 100).round(2).astype(str) + "%")

# ---------- 4) FEATURE/TARGET SETUP ----------
TARGET = "HighSatisfaction"
DROP_COLS = ["CustomerID"]  # Identifier, exclude from model input

X = df.drop(columns=[TARGET] + DROP_COLS)
y = df[TARGET].astype(int)

num_features = X.select_dtypes(include=np.number).columns.tolist()
cat_features = X.select_dtypes(exclude=np.number).columns.tolist()

print("\nNumerical features:", num_features)
print("Categorical features:", cat_features)

X_train, X_test, y_train, y_test = train_test_split(
    X, y,
    test_size=0.20,
    random_state=42,
    stratify=y
)

print("\nTrain shape:", X_train.shape, "Test shape:", X_test.shape)
print("Train target rate:", round(y_train.mean(), 4), "Test target rate:", round(y_test.mean(), 4))

# ---------- 5) PREPROCESSOR ----------
numeric_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="median")),
    ("scaler", StandardScaler())
])

categorical_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("onehot", OneHotEncoder(handle_unknown="ignore"))
])

preprocessor = ColumnTransformer(
    transformers=[
        ("num", numeric_transformer, num_features),
        ("cat", categorical_transformer, cat_features)
    ],
    remainder="drop"
)

# ---------- 6) HELPERS ----------
def find_best_threshold(y_true, y_prob):
    """Pick threshold maximizing F1 on provided data."""
    p, r, t = precision_recall_curve(y_true, y_prob)
    # precision_recall_curve returns thresholds with len = len(p)-1
    f1_scores = (2 * p[:-1] * r[:-1]) / (p[:-1] + r[:-1] + 1e-9)
    best_idx = int(np.argmax(f1_scores))
    return float(t[best_idx]), float(f1_scores[best_idx])

def evaluate_and_plot(model_name, model, X_test, y_test, threshold=0.5):
    """Evaluate metrics and plot confusion matrix heatmap."""
    y_prob = model.predict_proba(X_test)[:, 1]
    y_pred = (y_prob >= threshold).astype(int)

    metrics = {
        "Model": model_name,
        "Threshold": threshold,
        "Accuracy": accuracy_score(y_test, y_pred),
        "Precision": precision_score(y_test, y_pred, zero_division=0),
        "Recall": recall_score(y_test, y_pred, zero_division=0),
        "F1": f1_score(y_test, y_pred, zero_division=0),
        "ROC_AUC": roc_auc_score(y_test, y_prob),
        "PR_AUC": average_precision_score(y_test, y_prob)
    }

    print(f"\n===== {model_name} =====")
    for k, v in metrics.items():
        if k in ["Model"]:
            print(f"{k}: {v}")
        else:
            print(f"{k}: {v:.4f}")
    print("\nClassification Report:")
    print(classification_report(y_test, y_pred, digits=4))

    # Heatmap for each training model
    cm = confusion_matrix(y_test, y_pred)
    plt.figure(figsize=(5, 4))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues",
                xticklabels=["Pred 0", "Pred 1"],
                yticklabels=["Actual 0", "Actual 1"])
    plt.title(f"Confusion Matrix Heatmap - {model_name}")
    plt.ylabel("Actual")
    plt.xlabel("Predicted")
    plt.tight_layout()
    plt.show()

    return metrics, y_prob

def plot_top_permutation_importance(model_name, fitted_pipeline, X_eval, y_eval, top_n=15):
    """Permutation feature importance heatmap-friendly bar chart."""
    imp = permutation_importance(
        fitted_pipeline, X_eval, y_eval,
        scoring="f1", n_repeats=10, random_state=42, n_jobs=-1
    )
    importance_df = pd.DataFrame({
        "feature": X_eval.columns,
        "importance_mean": imp.importances_mean
    }).sort_values("importance_mean", ascending=False).head(top_n)

    plt.figure(figsize=(8, 6))
    sns.barplot(data=importance_df, x="importance_mean", y="feature", palette="viridis")
    plt.title(f"Top {top_n} Permutation Importances - {model_name}")
    plt.xlabel("Mean Importance (F1 drop)")
    plt.ylabel("Feature")
    plt.tight_layout()
    plt.show()

    return importance_df

# ---------- 7) MODEL 1: HUMAN-IMPLEMENTED ----------
# Logistic Regression (interpretable baseline)
human_model = Pipeline(steps=[
    ("preprocessor", preprocessor),
    ("classifier", LogisticRegression(
        max_iter=2000,
        class_weight="balanced",
        random_state=42
    ))
])

human_model.fit(X_train, y_train)
human_probs_train = human_model.predict_proba(X_train)[:, 1]
human_threshold, human_best_f1_train = find_best_threshold(y_train, human_probs_train)
print(f"\nHuman model train-optimal threshold: {human_threshold:.4f} (train F1={human_best_f1_train:.4f})")

human_metrics, _ = evaluate_and_plot(
    "Human Model - Logistic Regression",
    human_model, X_test, y_test, threshold=human_threshold
)

# ---------- 8) MODEL 2: AI-RECOMMENDED ----------
# Random Forest (non-linear interactions)
ai_model = Pipeline(steps=[
    ("preprocessor", preprocessor),
    ("classifier", RandomForestClassifier(
        n_estimators=400,
        max_depth=None,
        min_samples_split=5,
        min_samples_leaf=2,
        class_weight="balanced",
        random_state=42,
        n_jobs=-1
    ))
])

ai_model.fit(X_train, y_train)
ai_probs_train = ai_model.predict_proba(X_train)[:, 1]
ai_threshold, ai_best_f1_train = find_best_threshold(y_train, ai_probs_train)
print(f"\nAI model train-optimal threshold: {ai_threshold:.4f} (train F1={ai_best_f1_train:.4f})")

ai_metrics, _ = evaluate_and_plot(
    "AI Model - Random Forest",
    ai_model, X_test, y_test, threshold=ai_threshold
)
    
# ---------- 9) MODEL 3: REFINED ----------
# XGBoost with randomized search and imbalance handling
pos_weight = (y_train == 0).sum() / max((y_train == 1).sum(), 1)
print("\nscale_pos_weight:", round(float(pos_weight), 4))

refined_pipe = Pipeline(steps=[
    ("preprocessor", preprocessor),
    ("classifier", XGBClassifier(
        objective="binary:logistic",
        eval_metric="logloss",
        random_state=42,
        n_jobs=-1,
        scale_pos_weight=pos_weight
    ))
])

param_dist = {
    "classifier__n_estimators": [200, 300, 500],
    "classifier__max_depth": [3, 4, 5, 6],
    "classifier__learning_rate": [0.01, 0.03, 0.05, 0.1],
    "classifier__subsample": [0.7, 0.8, 0.9, 1.0],
    "classifier__colsample_bytree": [0.7, 0.8, 0.9, 1.0],
    "classifier__min_child_weight": [1, 3, 5],
    "classifier__gamma": [0, 0.1, 0.3]
}

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

search = RandomizedSearchCV(
    estimator=refined_pipe,
    param_distributions=param_dist,
    n_iter=25,
    scoring="f1",
    n_jobs=-1,
    cv=cv,
    verbose=1,
    random_state=42
)

search.fit(X_train, y_train)
best_refined_model = search.best_estimator_
print("\nBest refined params:")
print(search.best_params_)

refined_probs_train = best_refined_model.predict_proba(X_train)[:, 1]
refined_threshold, refined_best_f1_train = find_best_threshold(y_train, refined_probs_train)
print(f"\nRefined model train-optimal threshold: {refined_threshold:.4f} (train F1={refined_best_f1_train:.4f})")

refined_metrics, refined_probs_test = evaluate_and_plot(
    "Refined Model - Tuned XGBoost",
    best_refined_model, X_test, y_test, threshold=refined_threshold
)

# ---------- 10) MODEL 4: CATBOOST (ADDITIONAL ALTERNATIVE) ----------
# CatBoost usually performs strongly on mixed tabular datasets with categorical features.
catboost_model = CatBoostClassifier(
    loss_function="Logloss",
    eval_metric="AUC",
    random_state=42,
    verbose=0,
    auto_class_weights="Balanced",
    iterations=500,
    learning_rate=0.05,
    depth=6
)

catboost_model.fit(X_train, y_train, cat_features=cat_features)
catboost_probs_train = catboost_model.predict_proba(X_train)[:, 1]
catboost_threshold, catboost_best_f1_train = find_best_threshold(y_train, catboost_probs_train)
print(f"\nCatBoost train-optimal threshold: {catboost_threshold:.4f} (train F1={catboost_best_f1_train:.4f})")

catboost_metrics, _ = evaluate_and_plot(
    "Alternative Model - CatBoost",
    catboost_model, X_test, y_test, threshold=catboost_threshold
)

# ---------- 11) MODEL COMPARISON TABLE ----------
results_df = pd.DataFrame([human_metrics, ai_metrics, refined_metrics, catboost_metrics])
results_df = results_df.sort_values(by=["F1", "Recall", "PR_AUC"], ascending=False).reset_index(drop=True)

print("\n===== MODEL COMPARISON =====")
display(results_df.style.format({
    "Threshold": "{:.4f}",
    "Accuracy": "{:.4f}",
    "Precision": "{:.4f}",
    "Recall": "{:.4f}",
    "F1": "{:.4f}",
    "ROC_AUC": "{:.4f}",
    "PR_AUC": "{:.4f}"
}))

# Heatmap view of model metrics
plt.figure(figsize=(8, 4))
metric_cols = ["Accuracy", "Precision", "Recall", "F1", "ROC_AUC", "PR_AUC"]
heat_df = results_df.set_index("Model")[metric_cols]
sns.heatmap(heat_df, annot=True, fmt=".3f", cmap="YlGnBu")
plt.title("Performance Heatmap Across Models")
plt.tight_layout()
plt.show()

# Select final model for feature importance and export
model_registry = {
    "Human Model - Logistic Regression": human_model,
    "AI Model - Random Forest": ai_model,
    "Refined Model - Tuned XGBoost": best_refined_model,
    "Alternative Model - CatBoost": catboost_model
}
threshold_registry = {
    "Human Model - Logistic Regression": human_threshold,
    "AI Model - Random Forest": ai_threshold,
    "Refined Model - Tuned XGBoost": refined_threshold,
    "Alternative Model - CatBoost": catboost_threshold
}

# Force CatBoost as the final deployment model.
final_model_name = "Alternative Model - CatBoost"
final_model = model_registry[final_model_name]
final_threshold = threshold_registry[final_model_name]
print(f"\nSelected final deployment model: {final_model_name}")

# ---------- 12) FEATURE IMPORTANCE (FINAL MODEL) ----------
importance_df = plot_top_permutation_importance(
    final_model_name,
    final_model,
    X_test,
    y_test,
    top_n=15
)

print("\nTop features (permutation importance):")
print(importance_df.to_string(index=False))

        json.dump({
            "target": TARGET,
            "drop_cols": DROP_COLS,
            "num_features": num_features,
            "cat_features": cat_features,
            "model_name": model_key
        }, f, indent=2)

print("\nSaved artifacts in:", ARTIFACT_DIR)
print("- restaurant_satisfaction_model.joblib")
print("- feature_config.joblib")
print("- threshold.json")
print("- model_comparison_results.csv")
print("- final_model_feature_importance.csv")
print("\nSaved individual model files in:", INDIVIDUAL_DIR)
print("- logistic_regression.pkl + threshold/meta")
print("- random_forest.pkl + threshold/meta")
print("- xgboost_refined.pkl + threshold/meta")
print("- catboost.pkl + threshold/meta")
